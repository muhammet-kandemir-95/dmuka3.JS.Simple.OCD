dmuka3.JS.Simple.OCD Nedir?

 dmuka3.JS.Simple.OCD kütüphanesi sizin SEO ile tamamen entegre sitelerinizi kolay bir şekilde kodlamanızı sağlayan bir JavaScript kütüphanesidir.

 Dökümanın ilerleyen kısımlarında yer alan örneklerde de görüleceği üzere HTML verisine göre kodlar şekillenmektedir. Server Side Render ile kullanılması önerilmektedir.

İndir

Proje ile alakalı dosyaları indirebileceğiz kaynaklar bu kısımda yer almaktadır.

Önemli Not

Şemadaki root dışındaki tüm alt katmanlar aynı şemayı kullanmaktadır. root ise diğerlerinden farklı olarak parentQuery, single ve jobject parametrelerini kullanılamaz!

alias

alias verinin takma adı. 

<div id="app">
	<input type="text" class="name">
	<input type="text" class="surname">
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'name',
			single: true,
			query: '>.name'
		}, {
			alias: 'surname',
			single: true,
			query: '>.surname'
		}]
	});

	$app.name = 'Muhammet';
	$app.surname = 'Kandemir';
</script>

parentQuery

parentQuery eğer seçim yapılacak element birden fazla ise, kullanılması gerekmektedir. Aşağıdaki örnekte de görüldüğü üzere parentQuery eğer element ilgili bloğun hemen altında yer almıyorsa oldukça iş görmektedir. 

<style>
	span {
		display: inline-block;
		margin-right: 10px;
	}
</style>
<div id="app">
	<div>
		<span ocd-clone></span>
		<span>Data</span>
		<span>Data</span>
		<span>Data</span>
	</div>
	<br>
	<button>Add a New Data</button>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'datas',
			parentQuery: '>div',
			query: '>span'
		}, {
			alias: 'addNewData',
			single: true,
			query: '>button',
			on: {
				click: function (e) {
					this.$parent.datas.$add('New Data');
				}
			}
		}]
	});
</script>

Uyarı

parentQuery kullanılan yerde mutlaka klonlanacak olan element ocd-clone attribute'ü ile beraber bir tane olacak şekilde yer almalıdır. Örnekteki "<span ocd-clone>" gibi. Tabi dökümanın ilerleyen kısımlarında yer alan clone özelliğini override ederek bu durum değiştirilebilir. Fakat varsayılan durumda bu kural geçerlidir.

Peki eğer parentQuery için ayrı bir elementimiz yok ise? Hemen başka bir örneğe geçelim. 

<style>
	span {
		display: inline-block;
		margin-right: 10px;
	}
</style>
<div id="app">
	<button>Add a New Data</button>
	<br>
	<br>
	<span ocd-clone></span>
	<span>Data</span>
	<span>Data</span>
	<span>Data</span>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'datas',
			parentQuery: '',
			query: '>span'
		}, {
			alias: 'addNewData',
			single: true,
			query: '>button',
			on: {
				click: function (e) {
					this.$parent.datas.$add('New Data');
				}
			}
		}]
	});
</script>

Aslında görüldüğü üzere sadece parentQuery değerine '' değerini atamamız başka bir element için olan zorunluluğu ortadan kaldırıyor. Yukarıdaki örnekte parentQuery artık '#app' olarak kabul edilmektedir.

Ayrıca

parentQuery için ayrı bir elementin olması ve sadece içerisinde tekrar eden elementlerin bulunması önerilir. Yani ilk örnekteki yapı temiz projeler için daha uygun olmaktadır.

clone

clone parentQuery kullanılan yerlerde yeni bir satır oluşturulacağında kullanılan fonksiyon. Varsayılan değerinde parentQuery içerisinde yer alan ocd-clone attribute değerini taşıyan elemente göre çalışmaktadır. Aşağıdaki örnekte yeni oluşturulacak olan elementi manuel olarak oluşturuyoruz. 

<style>
	.red {
		background-color: red;
	}

	.green {
		background-color: green;
	}

	.blue {
		background-color: blue;
	}

	span {
		display: inline-block;
		margin-right: 10px;
		color: white;
		padding-left: 10px;
		padding-right: 10px;
	}
</style>
<div id="app">
	<div>
	</div>
	<br>
	<select>
		<option value="red">Red</option>
		<option value="green">Green</option>
		<option value="blue">Blue</option>
	</select>
	<button>Add a Row</button>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'items',
			parentQuery: '>div',
			query: '>span',
			clone: function (value) {
				var $el = $d.q.create('span');
				$el.$.attr('class', value);
				return $el;
			}
		}, {
			alias: 'select',
			single: true,
			query: '>select'
		}, {
			alias: 'button',
			single: true,
			query: '>button',
			on: {
				click: function (e) {
					this.$parent.items.$add(this.$parent.select);
				}
			}
		}]
	});
</script>

İleride daha da iyi görüleceği üzere parentQuery ve single verinin çıktısını Array ve Object olacak şekilde değiştirmektedir.

Aynı zamanda eğer single değeri true değil ise o zaman yazılan tüm komutlar her bir row için ayrı ayrı geçerli olacaktır.

query

query işlem yapılacak ana element'e ait bir bilgidir. Fakat burada önemli olan query değeri parentQuery veya single ile doğrudan ilişkilidir. Çünkü eğer single değeri true ise tek bir element, parentQuery değeri dolu ise birden fazla elementi temsil etmektedir. Aşağıda her iki örneği de inceleyeceğiz.

Şimdi parentQuery örneğine bakalım. 

<style>
	#app input {
		width: 100%;
	}
</style>
<div id="app">
	<input ocd-clone type="text" placeholder="Press enter to add a new line, press backspace to remove while empty">
	<input type="text" placeholder="Press enter to add a new line, press backspace to remove while empty">
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'inputs',
			parentQuery: '',
			query: '>input',
			on: {
				keydown: function (e) {
					if (e.keyCode === 13/*ENTER*/) {
						var $new = this.$parent.$insert(this.$index + 1, this.value);
						$new.$el.focus();
					} else if (e.keyCode === 8/*BACKSPACE*/ && this.value.length === 0 && this.$parent.length > 1) {
						e.preventDefault();

						var index = this.$index;
						this.$remove();
						this.$parent[Math.max(0, index - 1)].$el.focus();
					}
				}
			}
		}]
	});
</script>

Sanki beğendiğinizi duyuyor gibiyiz. Buradan sonra işler çok daha eğlenceli bir hale gelecek.

get

get sadece single değeri true iken çalışmaktadır. Amaç ise ilgili elementin değerinin alınmasını sağlamakdır. Aşağıdaki tablo üzerinde varsayılan değerler açıklanmıştır. Bu fonksiyonu override işlemine sokarak kendi değerinizi elde edebilirsiniz.

Element's Tag Name Default Value
INPUT[type=checkbox] $el.checked
INPUT[type=radio] $el.checked
INPUT $el.value
SELECT $el.value
ANY $el.innerHTML

ANY denilen kısım bunların dışında manasına gelmektedir. Eğer verinin kaynağı elementin sahip olduğu başka bir veriden gelecek ise override yaparak bunu sağlayabiliriz. Örnek set kısmında yer alıyor.

set

set sadece single değeri true iken çalışmaktadır. Amaç ise ilgili elementin değerinin atamasını sağlamakdır. Aşağıdaki tablo üzerinde varsayılan atanan değerler açıklanmıştır. Bu fonksiyonu override işlemine sokarak kendi değerinizin atamasını değiştirebilirsiniz.

Element's Tag Name Default Value
INPUT[type=checkbox] $el.checked
INPUT[type=radio] $el.checked
INPUT $el.value
SELECT $el.value
ANY $el.innerHTML

ANY denilen kısım bunların dışında manasına gelmektedir. Eğer verinin atanması sırasında başka bir işlem yapılacak ise override yaparak bunu sağlayabiliriz. Örneğe bakalım. 

<div id="app">
	<span>
		SPAN
	</span>
	<br>
	<br>
	<input type="text" placeholder="Enter span style">
	<br>
	<br>
	<div></div>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'span',
			single: true,
			query: '>span',
			get: function () {
				return this.$el.$.attr('style');
			},
			set: function (value) {
				this.$el.$.attr('style', value);
			}
		}, {
			alias: 'div',
			single: true,
			query: '>div'
		}, {
			alias: 'input',
			single: true,
			query: '>input',
			on: {
				input: function (e) {
					this.$parent.span = this.value;
					this.$parent.div = this.$parent.span;
				}
			}
		}]
	});
</script>

Ayrıca

watches event'i set işleminden sonra tetiklenmektedir.

data

data bir elemente ait birden fazla veri olabilmesini sağlamak amacıyla kullanılmaktadır. Bir metin kutusunun metin değeri dışında yazı rengi, arka plan rengi gibi ekstra özellikleri olmasını istiyorsanız data kullanmanız gerekmektedir.

data verisinde default yani varsayılan değer, değerin alınması için get fonksiyonu ve değerin doldurulması durumu için set fonksiyonu gerekmektedir.

default, get ve set değerlerinin varsayılan değerleri mevcuttur. Fakat hiçbiri olmadığında data sadece element üzerinde veri saklamak için olmuş olacaktır. O zaman veri sadece anlık sayfa için RAM üzerinde saklanacaktır.

data arka planda element üzerinde veri saklamaktadır. Fakat bu değerin değişmesi isteniyorsa set fonksiyonu içerisinde return ile değer döndürülmelidir. 

<div id="app">
	<input type="text" value="Muhammet Kandemir">
	<br>
	<br>
	<button class="change-value">Change Value</button>
	<button class="change-data-background-color">Change Background Color</button>
	<button class="change-data-color">Change Color</button>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'input',
			single: true,
			query: '>input',
			data: {
				backgroundColor: {
					default: 'green',
					get: function (key, currentValue) {
						return this.$el.$.css('background-color');
					},
					set: function (value, key) {
						this.$el.$.css('background-color', value);
					}
				},
				color: {
					default: 'black',
					get: function (key, currentValue) {
						return this.$el.$.css('color');
					},
					set: function (value, key) {
						this.$el.$.css('color', value);
					}
				}
			}
		}, {
			alias: 'changeValue',
			single: true,
			query: '>button.change-value',
			on: {
				click: function (e) {
					this.$parent.input = this.$parent.input.value.split('').reverse().join('');
				}
			}
		}, {
			alias: 'changeDataBackgroundColor',
			single: true,
			query: '>button.change-data-background-color',
			on: {
				click: function (e) {
					if (this.$parent.input.backgroundColor === 'green') {
						this.$parent.input.backgroundColor = 'blue';
					} else {
						this.$parent.input.backgroundColor = 'green';
					}
				}
			}
		}, {
			alias: 'changeDataColor',
			single: true,
			query: '>button.change-data-color',
			on: {
				click: function (e) {
					if (this.$parent.input.color === 'black') {
						this.$parent.input.color = 'white';
					} else {
						this.$parent.input.color = 'black';
					}
				}
			}
		}]
	});
</script>

Ayrıca

default ve get birlikte kullanıldığında ilk önce get çalıştırılır. Sonrasında gelen değer null, undefined veya '' ise set fonksiyonuna default değer gönderilir. Değil ise get fonksiyonundan gelen değer set fonksiyonuna gönderilir.

Bu durumda şunu anlamamız gerekmektedir; ilk yüklemede set mutlaka, get ve default duruma bağlı çalışmaktadır.

methods

methods geçerli element üzerinde belli başlı metotları tanımlandığı alandır. 

<div id="app">
	<input name="name" type="text" value="Muhammet" placeholder="Name">
	<br>
	<br>
	<input name="surname" type="text" value="Kandemir" placeholder="Surname">
	<br>
	<br>
	<input name="class" type="text" value="P-95" placeholder="Class">
	<br>
	<br>
	<button>Clear Form</button>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'name',
			single: true,
			query: '>input[name="name"]'
		}, {
			alias: 'surname',
			single: true,
			query: '>input[name="surname"]'
		}, {
			alias: 'class',
			single: true,
			query: '>input[name="class"]'
		}, {
			alias: 'clear',
			single: true,
			query: '>button',
			methods: {
				clearForm: function () {
					this.$parent.$set({
						name: '',
						surname: '',
						class: ''
					});
				}
			},
			on: {
				click: function (e) {
					this.clearForm();
				}
			}
		}]
	});
</script>

Örnekte de görüldüğü üzere methods kullanılabilir. clearForm içerisinde arguments veya return kullanılabilir. Başka bir örnekte bunu inceleyelim. 

<div id="app">
	<input name="number1" type="text" value="20" placeholder="Number 1">
	<br>
	<br>
	<input name="number2" type="text" value="10" placeholder="Number 2">
	<br>
	<br>
	<button>Show Sum</button>
</div>

<script src="dmuka3.JS.Simple.OCD.min.js"></script>
<script>
	var $app = $d.ocd({
		query: '#app',
		sub: [{
			alias: 'number1',
			single: true,
			query: '>input[name="number1"]'
		}, {
			alias: 'number2',
			single: true,
			query: '>input[name="number2"]'
		}, {
			alias: 'clear',
			single: true,
			query: '>button',
			methods: {
				sum: function (a, b) {
					return parseFloat(a) + parseFloat(b);
				}
			},
			on: {
				click: function (e) {
					var result = this.sum(this.$parent.number1, this.$parent.number2);
					alert('Sum : ' + result);
				}
			}
		}]
	});
</script>
